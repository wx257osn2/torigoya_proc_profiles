# add some options for clang 3.5 and later

overwrite:
  compile:
    allowed_command_line:
      "-std=": { select: [ "c++11", "gnu++11", "c++98", "gnu++98" ] }

push_back:
  compile:
    allowed_command_line:
      "-std=": { select: [ "c++1z", "gnu++1z", "c++14", "gnu++14" ] }
      "-fstack-protector": { select: [ "-strong" ] }
